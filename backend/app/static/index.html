OTP API Entegrasyon Kılavuzu:

curl -X POST http://localhost:8000/api/v1/send-otp \
  -H "X-API-Key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"recipient":"905551234567","message_text":"Doğrulama kodunuz: 629481"}'
import requests
 
url = "http://localhost:8000/api/v1/send-otp"
headers = {
    "X-API-Key": "YOUR_API_KEY",
    "Content-Type": "application/json"
}
payload = {
    "recipient": "905551234567",
    "message_text": "Doğrulama kodunuz: 629481"
}
 
response = requests.post(url, headers=headers, json=payload)
print(response.json())
const axios = require('axios');
 
const url = 'http://localhost:8000/api/v1/send-otp';
const headers = {
  'X-API-Key': 'YOUR_API_KEY',
  'Content-Type': 'application/json'
};
const payload = {
  recipient: '905551234567',
  message_text: 'Doğrulama kodunuz: 629481'
};
 
axios.post(url, payload, { headers })
  .then(res => console.log(res.data))
  .catch(err => console.error(err.response ? err.response.data : err.message));
<?php
 
$url = 'http://localhost:8000/api/v1/send-otp';
$headers = [
    'X-API-Key: YOUR_API_KEY',
    'Content-Type: application/json'
];
$payload = [
    'recipient' => '905551234567',
    'message_text' => 'Doğrulama kodunuz: 629481'
];
 
$ch = curl_init($url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_HTTPHEADER, $headers);
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($payload));
 
$response = curl_exec($ch);
curl_close($ch);
 
echo $response;
